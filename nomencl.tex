\DeclareAcronym{unit}{
  short = \ensuremath{()} ,
  long  = The unit value returned by computation without result,
  sort  = unit,
  first-style = short,
  class = nomencl
}
\DeclareAcronym{prop}{
  short = \ensuremath{\mathbb{P}} ,
  long  = The type of logic statements which can either be true or false,
  sort  = prop,
  first-style = short,
  class = nomencl
}

  sort  = powerset,
  first-style = short,
  class = nomencl
}
